        private SQLiteConnection sql_con;
        private SQLiteCommand sql_cmd;
        private SQLiteDataAdapter DA;
        private DataSet DS = new DataSet();
        private DataTable DT = new DataTable();

        private void SetConnection()
        {
            sql_con = new SQLiteConnection("Data Source=MotorradDB.db;Version=3;New=False;Compress=true;");
        }

        private void LoadData()
        {
            SetConnection();
            sql_con.Open();
            sql_cmd = sql_con.CreateCommand();
            string CommandText = "select * from MotorradTB_Clientes";
            DA = new SQLiteDataAdapter(CommandText, sql_con);
            DS.Reset();
            DA.Fill(DS);
            DT = DS.Tables[0];
            DataGrid_Clientes.DataSource = DT;
            sql_con.Close();
        }
        private void ExecuteQuery(string txtQuery)
        {
            SetConnection();
            sql_con.Open();
            sql_cmd = sql_con.CreateCommand();
            sql_cmd.CommandText = txtQuery;
            sql_cmd.ExecuteNonQuery();
            sql_con.Close();
        }

        public void DisplayValue()
        {
            SetConnection();
            DataTable DT;

            sql_con.Open();

            DA = new SQLiteDataAdapter("SELECT * from MotorradTB_Clientes", sql_con);
            DT = new DataTable();
            DA.Fill(DT);
            DataGrid_Clientes.DataSource = DT;
            sql_con.Close();
        }
        private void Txt_Pesquisar_TextChanged(object sender, EventArgs e)
        {

        }
        public void SearchData(string search)
        {
            SetConnection();
            string query = "SELECT * FROM MotorradTB_Clientes WHERE Nome like '%" + search + "%' or ID like '%" + search + "%' or Email like '%" + search + "%' or Celular like '%" + search + "%' or Fixo like '%" + search + "%' or Endereco like '%" + search + "%' or Numero like '%" + search + "%' or DataNascimento like '%" + search + "%' or DataCadastro like '%" + search + "%'";
            DA = new SQLiteDataAdapter(query, sql_con);
            DT = new DataTable();
            DA.Fill(DT);
            DataGrid_Clientes.DataSource = DT;
            sql_con.Close();
        }
